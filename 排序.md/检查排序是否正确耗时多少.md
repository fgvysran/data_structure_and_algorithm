import java.lang.reflect.Method;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) throws Exception {
        cheShi();
    }
    public static void cheShi() throws Exception {
        System.out.println("请输入要测试的数据大小");
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[]arr = new int[n];
        for(int i = 0;i < n;i++){
            arr[i] = (int)(Math.random() * 1_000_000_0);
        }//获取随机数据
        runAndCheck(Main.class.getMethod("merge_sort", int[].class, int.class, int.class), arr, 0, n - 1);
    }
    public static void runAndCheck(Method method, int[] arr, int l, int r) throws Exception {
        long start = System.currentTimeMillis();//记录开始时间
        method.invoke(null, arr, l, r); // 执行静态方法
        long end = System.currentTimeMillis();//结束时间

        if (check(arr, l, r)) {
            System.out.println("排序结果正确");
        } else {
            System.out.println("排序结果错误");
        }

        System.out.println(method.getName() + "用时: " + (end - start) + "ms");
    }

    public static boolean check(int[] arr, int l, int r){
        //[l,r]
        for(int i = l;i < r;i++){
            if(arr[i] > arr[i + 1])return false;
        }return true;
    }

    public static void merge_sort(int[] arr, int l, int r){//(l-r)
        if(l >= r)return;
        int mid = l + ((r - l) >> 1);
        merge_sort(arr,l,mid);
        merge_sort(arr,mid + 1,r);
        merge(arr,l,r,mid);
    }
    public static void merge(int[] arr, int l, int r, int mid){
        int[]temp = new int[r - l + 1];
        int x = l,y = mid + 1,c = 0;
        while(x <= mid && y <= r){
            temp[c++] = arr[x] <= arr[y]?arr[x++]:arr[y++];
        }
        while(x <= mid){
            temp[c++] = arr[x++];
        }
        while(y <= r){
            temp[c++] = arr[y++];
        }
        if (c >= 0) System.arraycopy(temp, 0, arr, l, c);
    }
}
